# Variables Options

<NixOption
  path={"variables"}
  type={"attribute set of (submodule)"}
  defaultValue={"{ }"}
  exampleCode={"{\n  # Shared config (plaintext, NOT encrypted)\n   \"/var/LOG_LEVEL\" = { value = \"info\"; };\n   \"/var/API_VERSION\" = { value = \"v1\"; };\n\n  # Secret (value lives in vars/postgres-url.sops.yaml)\n   \"/secret/postgres-url\" = { value = \"\"; };\n }"}
  module={"variables"}
>
Workspace variables keyed by their full variable ID.

Prefixes determine storage:
  /var/*      - Shared config (plaintext, NOT encrypted)
  /secret/*   - Flat secrets (one SOPS file per variable)
  /computed/* - Nix-computed values (read-only)

Secret variable values are empty strings; the SOPS file is the source of truth.
Plaintext variables store their value directly.
</NixOption>

<NixOption
  path={"variables.<name>.id"}
  type={"string"}
  defaultValue={"\"‹name›\""}
  module={"variables"}
>
Variable identifier. Format: //

This defaults to the attribute key, so it normally does not need to
be written in config files.

Secret variables use a flat namespace:
  /secret/postgres-url -\> .stack/secrets/vars/postgres-url.sops.yaml

Computed variables use /computed/\<source\>/\<path\>:
  /computed/apps/web/port
  /computed/services/postgres/port
</NixOption>

<NixOption
  path={"variables.<name>.isComputed"}
  type={"boolean"}
  readonly
  module={"variables"}
>
Whether this is a computed (read-only) variable
</NixOption>

<NixOption
  path={"variables.<name>.isPlaintext"}
  type={"boolean"}
  readonly
  module={"variables"}
>
Whether this is a plaintext config variable (/var/*)
</NixOption>

<NixOption
  path={"variables.<name>.isSecret"}
  type={"boolean"}
  readonly
  module={"variables"}
>
Whether this is a SOPS-encrypted secret
</NixOption>

<NixOption
  path={"variables.<name>.isValsRef"}
  type={"boolean"}
  readonly
  module={"variables"}
>
Whether the value is a vals reference
</NixOption>

<NixOption
  path={"variables.<name>.keyGroup"}
  type={"string"}
  readonly
  module={"variables"}
>
Key group extracted from ID (e.g., 'secret', 'var', 'computed')
</NixOption>

<NixOption
  path={"variables.<name>.secretYamlKey"}
  type={"null or string"}
  readonly
  module={"variables"}
>
Deterministic YAML key used inside the per-variable SOPS file
</NixOption>

<NixOption
  path={"variables.<name>.sopsFile"}
  type={"null or string"}
  readonly
  module={"variables"}
>
Path to the SOPS file for this keygroup (null for /var/* and /computed/*)
</NixOption>

<NixOption
  path={"variables.<name>.value"}
  type={"string"}
  defaultValue={"\"\""}
  module={"variables"}
>
The value of this variable.

For plaintext variables (/var/*): the literal value.
For secrets (/secret/*): empty string (SOPS file is source of truth).
For computed (/computed/*): the computed value from Nix.

Legacy: ref+sops:// values are still supported during migration.
</NixOption>

<NixOption
  path={"variables.<name>.varName"}
  type={"string"}
  readonly
  module={"variables"}
>
Variable name extracted from ID (last path component)
</NixOption>
