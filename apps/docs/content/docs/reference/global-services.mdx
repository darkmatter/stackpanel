# Global-services Options

<NixOption
  path={"globalServices.caddy.enable"}
  type={"boolean"}
  defaultValue={"false"}
  exampleCode={"true"}
  module={"global-services"}
>
Whether to enable Global Caddy reverse proxy.
</NixOption>

<NixOption
  path={"globalServices.caddy.sites"}
  type={"attribute set of string"}
  defaultValue={"{ }"}
  exampleCode={"{\n  \"api.localhost\" = \"localhost:8080\";\n  \"myapp.localhost\" = \"localhost:3000\";\n}"}
  module={"global-services"}
>
Sites to register with Caddy (domain -\> upstream)
</NixOption>
