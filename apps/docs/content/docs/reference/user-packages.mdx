# User-packages Options

<NixOption
  path={"userPackages.enable"}
  type={"boolean"}
  defaultValue={"true"}
  exampleCode={"true"}
  module={"user-packages"}
>
Whether to enable user-installed packages from .stack/data/packages.nix.
</NixOption>

<NixOption
  path={"userPackages.packages"}
  type={"list of string"}
  defaultCode={"[\n  \"oxlint\"\n  \"oxfmt\"\n]"}
  readonly
  module={"user-packages"}
>
List of user-installed package attribute paths (read from packages.nix)
</NixOption>

<NixOption
  path={"userPackages.resolved"}
  type={"list of package"}
  readonly
  module={"user-packages"}
>
Resolved package derivations from user package list.

Only populated when pkgs is available. Empty list otherwise.
</NixOption>

<NixOption
  path={"userPackages.serialized"}
  type={"list of (attribute set)"}
  readonly
  module={"user-packages"}
>
Serialized package info for config JSON.

Only populated when pkgs is available. Empty list otherwise.
</NixOption>
