# Healthchecks Options

<NixOption
  path={"healthchecks.defaultInterval"}
  type={"null or signed integer"}
  defaultValue={"30"}
  module={"healthchecks"}
>
Default interval for automatic healthcheck runs in seconds.
Set to null to disable automatic checks.
</NixOption>

<NixOption
  path={"healthchecks.defaultTimeout"}
  type={"signed integer"}
  defaultValue={"10"}
  module={"healthchecks"}
>
Default timeout for healthchecks in seconds
</NixOption>

<NixOption
  path={"healthchecks.enable"}
  type={"boolean"}
  defaultValue={"true"}
  module={"healthchecks"}
>
Enable the healthchecks system
</NixOption>

<NixOption
  path={"healthchecks.modules"}
  type={"attribute set of (submodule)"}
  defaultValue={"{ }"}
  exampleCode={"{\n  go = {\n    displayName = \"Go\";\n    checks = {\n      # Inline script\n      go-installed = {\n        script = \"command -v go >/dev/null 2>&1\";\n        severity = \"critical\";\n      };\n    };\n  };\n\n  postgres = {\n    displayName = \"PostgreSQL\";\n    checks = {\n      # TCP check\n      running = {\n        type = \"tcp\";\n        tcpHost = \"localhost\";\n        tcpPort = 5432;\n        severity = \"critical\";\n      };\n      # Path to script file\n      has-tables = {\n        path = ./.stack/src/checks/postgres/has-tables.sh;\n        severity = \"warning\";\n      };\n    };\n  };\n\n  # Extension module\n  sst = {\n    displayName = \"SST\";\n    checks = {\n      configured = {\n        path = ./src/checks/configured.sh;\n        severity = \"critical\";\n      };\n    };\n  };\n}"}
  module={"healthchecks"}
>
Healthchecks organized by module.

Each module can declare multiple healthchecks that verify its
functionality. The UI displays a traffic light for each module
based on the aggregate status of its healthchecks.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks"}
  type={"attribute set of (submodule)"}
  defaultValue={"{ }"}
  module={"healthchecks"}
>
Healthchecks for this module
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.description"}
  type={"null or string"}
  defaultValue={"null"}
  module={"healthchecks"}
>
Description of what this check verifies
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.enable"}
  type={"boolean"}
  defaultValue={"true"}
  module={"healthchecks"}
>
Whether this healthcheck is enabled
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.httpExpectedStatus"}
  type={"signed integer"}
  defaultValue={"200"}
  module={"healthchecks"}
>
Expected HTTP status code for a healthy response
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.httpMethod"}
  type={"string"}
  defaultValue={"\"GET\""}
  module={"healthchecks"}
>
HTTP method to use for http-type checks
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.httpUrl"}
  type={"null or string"}
  defaultValue={"null"}
  exampleCode={"\"http://localhost:3000/health\""}
  module={"healthchecks"}
>
URL to check for http-type checks
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.interval"}
  type={"null or signed integer"}
  defaultValue={"null"}
  module={"healthchecks"}
>
How often to run this check in seconds.
If null, check runs only on demand.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.name"}
  type={"string"}
  defaultValue={"\"‹name›\""}
  module={"healthchecks"}
>
Display name for the healthcheck
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.nixExpr"}
  type={"null or string"}
  defaultValue={"null"}
  exampleCode={"\"builtins.pathExists /nix/store\""}
  module={"healthchecks"}
>
Nix expression to evaluate for nix-type checks.
Should evaluate to true for healthy, false for unhealthy.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.path"}
  type={"null or absolute path"}
  defaultValue={"null"}
  exampleCode={"./.stack/src/checks/postgres/can-connect.sh"}
  module={"healthchecks"}
>
Path to script file for script-type checks.
Content is read and used as the script body.
Mutually exclusive with `script` and `scriptRef`.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.script"}
  type={"null or string"}
  defaultValue={"null"}
  exampleCode={"\"command -v go >/dev/null 2>&1\""}
  module={"healthchecks"}
>
Shell script content for script-type checks (inline).
Should exit 0 for healthy, non-zero for unhealthy.
Mutually exclusive with `path` and `scriptRef`.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.scriptPackage"}
  type={"null or package"}
  defaultValue={"null"}
  module={"healthchecks"}
>
A derivation that provides the healthcheck script.
The script should be at $out/bin/ or $out.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.scriptRef"}
  type={"null or string"}
  defaultValue={"null"}
  exampleCode={"\"db-connect\""}
  module={"healthchecks"}
>
Reference to a stackpanel.scripts.* entry.
The referenced script is used as the healthcheck command.
Mutually exclusive with `script` and `path`.
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.severity"}
  type={"one of \"critical\", \"warning\", \"info\""}
  defaultValue={"\"warning\""}
  module={"healthchecks"}
>
How critical this check is:
- critical: Failing = unhealthy (red light)
- warning: Failing = degraded (yellow light)
- info: Informational only, doesn't affect status
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.tags"}
  type={"list of string"}
  defaultValue={"[ ]"}
  exampleCode={"[\n  \"database\"\n  \"critical\"\n]"}
  module={"healthchecks"}
>
Tags for filtering/grouping checks
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.tcpHost"}
  type={"null or string"}
  defaultValue={"null"}
  exampleCode={"\"localhost\""}
  module={"healthchecks"}
>
Host to connect to for tcp-type checks
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.tcpPort"}
  type={"null or signed integer"}
  defaultValue={"null"}
  exampleCode={"5432"}
  module={"healthchecks"}
>
Port to connect to for tcp-type checks
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.timeout"}
  type={"signed integer"}
  defaultValue={"10"}
  module={"healthchecks"}
>
Timeout for the check in seconds
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.checks.<name>.type"}
  type={"one of \"script\", \"nix\", \"http\", \"tcp\""}
  defaultValue={"\"script\""}
  module={"healthchecks"}
>
Type of healthcheck (script, nix, http, tcp)
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.displayName"}
  type={"string"}
  defaultValue={"\"‹name›\""}
  module={"healthchecks"}
>
Display name for the module in the UI
</NixOption>

<NixOption
  path={"healthchecks.modules.<name>.enable"}
  type={"boolean"}
  defaultValue={"true"}
  module={"healthchecks"}
>
Enable healthchecks for this module
</NixOption>

<NixOption
  path={"healthchecksComputed"}
  type={"attribute set of attribute set of unspecified value"}
  defaultValue={"{ }"}
  readonly
  module={"healthchecks"}
>
Computed healthcheck configurations (for agent API)
</NixOption>

<NixOption
  path={"healthchecksList"}
  type={"list of unspecified value"}
  defaultValue={"[ ]"}
  readonly
  module={"healthchecks"}
>
Flat list of all healthchecks (for agent API)
</NixOption>
